    // 打乱选项内容顺序
    function shuffleOptions(options) {
        const shuffled = [...options];
        for (let i = shuffled.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [shuffled[i], shuffled[j]] = [shuffled[j], shuffled[i]];
        }
        return shuffled;
    }

    // 随机挑选题目
    function selectRandomQuestions(questionBank, count, category) {
        const questions = [];
        const availableQuestions = [...questionBank[category]];
        for (let i = 0; i < count; i++) {
            const randomIndex = Math.floor(Math.random() * availableQuestions.length);
            questions.push(availableQuestions[randomIndex]);
            availableQuestions.splice(randomIndex, 1); // 避免重复
        }
        return questions;
    }

    // 渲染题目到页面
    function renderQuestions(selectedQuestions) {
        selectedQuestions.forEach((q, index) => {
            const shuffledOptions = shuffleOptions(q.options);
            let optionsHTML = "";
            shuffledOptions.forEach((option, idx) => {
                const letter = String.fromCharCode(65 + idx); // A, B, C, D
                optionsHTML += `<label><input type="radio" name="q${index + 1}" value="${option.value}"> ${letter}. ${option.text}</label><br>`;
            });
            document.getElementById(`q${index + 1}`).innerHTML = `<p>${q.question}</p>${optionsHTML}`;
        });
    }

    // 初始化测验
    const selectedResponsibilityQuestions = selectRandomQuestions(questionBank, 10, "responsibility");
    const selectedIQQuestions = selectRandomQuestions(questionBank, 5, "iq");
    const selectedQuestions = [...selectedResponsibilityQuestions, ...selectedIQQuestions];
    renderQuestions(selectedQuestions);

    // 提交测验
    function submitQuiz() {
        // 获取候选人姓名
        const candidateName = document.getElementById("name").value.trim();
        if (!candidateName) {
            alert("请填写您的姓名！");
            return;
        }

        let score = 0;
        const answers = [];
        const correctAnswers = [];
        const explanations = [];

        // 责任心与担当部分（每题6分）
for (let i = 1; i <= 10; i++) {
    const selected = document.querySelector(`input[name=q${i}]:checked`);
    if (selected) {
        score += parseInt(selected.value); // 每题6分
        answers.push(selected.parentElement.textContent.trim());
    } else {
        answers.push("未作答");
    }
    correctAnswers.push(selectedQuestions[i - 1].options.find(option => option.value === 6)?.text || "");
    explanations.push(selectedQuestions[i - 1].explanation || "");
}

// 智商测试部分（每题8分）
for (let i = 11; i <= 15; i++) {
    const selected = document.querySelector(`input[name=q${i}]:checked`);
    if (selected) {
        score += parseInt(selected.value); // 每题8分
        answers.push(selected.parentElement.textContent.trim());
    } else {
        answers.push("未作答");
    }
    correctAnswers.push(selectedQuestions[i - 1].options.find(option => option.value === 8)?.text || "");
    explanations.push(selectedQuestions[i - 1].explanation || "");
}

        // 评测分析
        let analysis = "";
        if (score >= 90) {
            analysis = "优秀：您具备极强的责任心和出色的智商表现，非常适合担任制造科室主管。";
        } else if (score >= 70) {
